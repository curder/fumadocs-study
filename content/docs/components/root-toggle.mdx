## 用法 [#usage]

可以将此组件添加到侧边栏或所需的其他位置。

```tsx title="src/app/docs/layout.tsx"
import { RootToggle } from "fumadocs-ui/components/layout/root-toggle"; // [!code ++]

<DocsLayout
  // [!code ++:18]
  sidebar={{
    banner: (
      <RootToggle
        options={[
          {
            title: "Folder 1",
            description: "Pages in folder 1",
            url: "/docs/image",
          },
          {
            title: "Folder 2",
            description: "Pages in folder 2",
            url: "/docs",
          },
        ]}
      />
    ),
  }}
>
  {children}
</DocsLayout>;
```
